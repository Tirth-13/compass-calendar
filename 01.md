#include <vector>
#include <iostream>
using namespace std;
class Solution
{
public:
    int majorityElement(vector<int> &nums)
    {
        int n = nums.size();
        int i = 0;
        while (i < n - 1)
        {
            if (nums[i] == nums[i + 1])
            {
                i++;
            }
            else
            {
                nums.erase(nums.begin() + i);
                nums.erase(nums.begin() + i + 1);
                i=0;
            }
        }
        return nums[0];
    }
};
int main()
{
    Solution s;
    vector<int> v{2, 2, 3, 3, 3,2,2};
    int c = s.majorityElement(v);
    cout << c;
}
